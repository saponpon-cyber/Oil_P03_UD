// ===== Google Apps Script (รองรับ รูปภาพ → Google Drive) ===== const SHEET_ID = "YOUR_SPREADSHEET_ID"; const PM_EMAIL = "pm@yourcompany.com"; const DRIVE_FOLDER = "ระบบน้ำมัน_รูปภาพ"; // ชื่อ folder ใน Drive const SHEET_REQ = "ใบขอเบิก"; const SHEET_REC = "บันทึกน้ำมัน"; const SHEET_SYNC = "AppSync"; function doGet(e) { if (e.parameter.action === "load") return loadData(); return respond({status:"ok",msg:"Fuel API Ready"}); } function doPost(e) { try { const d = JSON.parse(e.postData.contents); if(d.action==="request") return saveRequest(d); if(d.action==="record") return saveRecord(d); if(d.action==="sync") return syncData(d); return respond({status:"error",msg:"unknown"}); } catch(err){ return respond({status:"error",msg:err.toString()}); } } // อัพรูปขึ้น Google Drive → return links function uploadPhotos(photos, subFolder) { if (!photos || photos.length === 0) return []; const root = getDriveFolder(DRIVE_FOLDER); const folder = getDriveFolder(subFolder, root); return photos.map((p, i) => { const blob = Utilities.newBlob( Utilities.base64Decode(p.data), p.mimeType, p.name || `photo_${i+1}.jpg` ); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return file.getUrl(); }); } function getDriveFolder(name, parent) { const src = parent || DriveApp; const it = src.getFoldersByName(name); return it.hasNext() ? it.next() : src.createFolder(name); } function saveRequest(d) { // อัพรูปขึ้น Drive const photoLinks = uploadPhotos(d.photos, `${d.plateNo}_${d.fuelDate}`); const photoCell = photoLinks.join('\n'); const sh = getOrCreate(SHEET_REQ,[ "ID","วันที่ขอ","วันที่เติม","ทะเบียน","ประเภท", "ลิตร","หน่วยงาน","ผู้ขับ","มิเตอร์เริ่ม", "มิเตอร์สิ้นสุด","ระยะ/ชม.","อัตราสิ้นเปลือง","สถานะเรท","เหตุผล","รูปภาพ" ]); sh.appendRow([d.id, new Date(), d.fuelDate, d.plateNo, d.machineType, d.liters, d.project, d.driverName, d.meterStart, d.meterEnd, d.distOrHr, d.consumeRate, d.rateStatus, d.reason, photoCell]); MailApp.sendEmail(PM_EMAIL, `📋 ขอเบิก: ${d.machineType} ${d.plateNo} — ${d.liters} ล. (${d.fuelDate})`, `${d.machineType} ${d.plateNo}\n${d.liters} ลิตร\nวันที่: ${d.fuelDate}\nผู้ขับ: ${d.driverName}\nมิเตอร์: ${d.meterStart} → ${d.meterEnd}\nระยะ: ${d.distOrHr}\nอัตรา: ${d.consumeRate}\n\nรูปภาพ:\n${photoLinks.join('\n')}`); return respond({status:"ok", photoLinks}); } function saveRecord(d) { const sh = getOrCreate(SHEET_REC,[ "Timestamp","RequestID","วันที่เติม","ทะเบียน","ประเภท", "ลิตรจริง","ราคา/ลิตร","รวมบาท","มิเตอร์เริ่ม","มิเตอร์สิ้นสุด", "ระยะ/ชม.","อัตราสิ้นเปลือง","สถานะเรท","ผู้ขับ","ผู้บันทึก","หน่วยงาน" ]); sh.appendRow([new Date(), d.requestId, d.fuelDate, d.plateNo, d.machineType, d.actualLiters, d.pricePerLiter, d.totalCost, d.meterStart, d.meterEnd, d.distOrHr, d.consumeRate, d.rateStatus, d.driverName, d.recorderName, d.project]); if(d.rateStatus==="เกินเรท") MailApp.sendEmail(PM_EMAIL, `⚠️ เกินเรท: ${d.machineType} ${d.plateNo} — ${d.consumeRate}`, `ทะเบียน: ${d.plateNo}\nอัตรา: ${d.consumeRate}\nวันที่: ${d.fuelDate}`); return respond({status:"ok"}); } function loadData() { const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ss.getSheetByName(SHEET_SYNC); if(!sh) return respond({status:"ok",records:[],pending:[],stock:[]}); const data = sh.getRange(1,1).getValue(); if(!data) return respond({status:"ok",records:[],pending:[],stock:[]}); return respond({status:"ok", ...JSON.parse(data)}); } function syncData(d) { const ss = SpreadsheetApp.openById(SHEET_ID); let sh = ss.getSheetByName(SHEET_SYNC); if(!sh){ sh = ss.insertSheet(SHEET_SYNC); sh.hideSheet(); } sh.getRange(1,1).setValue(JSON.stringify({ records:d.records||[], pending:d.pending||[], stock:d.stock||[] })); return respond({status:"ok"}); } function getOrCreate(name, headers) { const ss = SpreadsheetApp.openById(SHEET_ID); let sh = ss.getSheetByName(name); if(!sh){ sh = ss.insertSheet(name); sh.appendRow(headers); sh.getRange(1,1,1,headers.length).setBackground("#0f2744").setFontColor("#fff").setFontWeight("bold"); sh.setFrozenRows(1); } return sh; } function respond(obj){ return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }

Deploy → Web App
▶

1. กด Deploy → New deployment

2. Type: Web app | Execute as: Me | Access: Anyone

3. กด Deploy → คัดลอก URL → วางในช่องด้านบน

▶

2. อัพโหลด index.html ไฟล์นี้

3. Settings → Pages → Branch: main → Save

4. URL: https://username.github.io/fuel-system
